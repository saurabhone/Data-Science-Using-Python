column_names = {'Date': 'date',
                'Unamed: 2': 'boxscore', 'Opp': 'away_team',
                'W/L':'Win-Loss',
                'R': 'away_points', 'Tm': 'home_team',
                'RA': 'home_points'}

games = (games.rename(columns=column_names)
    .dropna(thresh=4)                    
    [['date', 'away_team', 'away_points', 'home_team', 'home_points','Win-Loss']]

#Now we will make the data set tidy using pandas.melt function
#This function is useful to massage a DataFrame into a format where one or more columns are 
#identifier variables(id_vars), while all other columns, considered measured variables (value_vars), 
#are “unpivoted” to the row axis, leaving just two non-identifier columns, ‘variable’ and ‘value’.

tidy = pd.melt(games.reset_index(),
               id_vars=['game_id', 'date','Win-Loss'], value_vars=['away_team', 'home_team'],
               value_name='team')
tidy.head()
