column_names = {'Date': 'date', 'Start (ET)': 'start',
                'Unamed: 2': 'box', 'Visitor/Neutral': 'away_team', 
                'PTS': 'away_points', 'Home/Neutral': 'home_team',
                'PTS.1': 'home_points', 'Unamed: 7': 'n_ot'}

 
#We have created a dictionary with key value pairs of current column names and new column names of table 'games'
#Now we will use some functions of pandas such as,
#.dropna : it has a thresh function which will drop if minimum NAN values are found in a row (In our example it is 4)
#.rename : to rename the columns with the dictionary keys and values
#.assign : it will assign new columns to a DataFrame and will return a new object with all the original columns in addition to the new ones.
#.set_index : it will set an index and the boolean will append columns to existing index
#.rename_axis : it will alter the axis and / or columns using input functions 
#.sort_index : it will sort objects by labels on the axis



games = (games.rename(columns=column_names)
    .dropna(thresh=4)
    [['date', 'away_team', 'away_points', 'home_team', 'home_points']]
    .assign(date=lambda x: pd.to_datetime(x['date'], format='%a, %b %d, %Y'))
    .set_index('date', append=True)
    .rename_axis(["game_id", "date"])
    .sort_index())
games.head()

#Now we will make the data set tidy using pandas.melt function
#This function is useful to massage a DataFrame into a format where one or more columns are 
#identifier variables(id_vars), while all other columns, considered measured variables (value_vars), 
#are “unpivoted” to the row axis, leaving just two non-identifier columns, ‘variable’ and ‘value’.
